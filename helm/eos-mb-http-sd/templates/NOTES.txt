1. Get the application URL by running these commands:
{{- if .Values.eosMbHttpSd.ingress.enabled }}
{{- range $host := .Values.eosMbHttpSd.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.eosMbHttpSd.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.eosMbHttpSd.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "eos-mb-http-sd.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.eosMbHttpSd.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "eos-mb-http-sd.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "eos-mb-http-sd.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.eosMbHttpSd.service.port }}
{{- else if contains "ClusterIP" .Values.eosMbHttpSd.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "eos-mb-http-sd.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Get the Prometheus URL by running these commands:
{{- if .Values.prometheus.enabled }}
{{- if .Values.prometheus.server.ingress.enabled }}
{{- range $host := .Values.prometheus.server.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.prometheus.server.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.prometheus.server.service.type }}
  export PROMETHEUS_NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Release.Name }}-prometheus-server)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$PROMETHEUS_NODE_PORT
{{- else if contains "LoadBalancer" .Values.prometheus.server.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ .Release.Name }}-prometheus-server'
  export PROMETHEUS_SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ .Release.Name }}-prometheus-server --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$PROMETHEUS_SERVICE_IP:{{ .Values.prometheus.server.service.port }}
{{- else if contains "ClusterIP" .Values.prometheus.server.service.type }}
  export PROMETHEUS_POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name=prometheus,app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export PROMETHEUS_CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $PROMETHEUS_POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:9090 to use Prometheus"
  kubectl --namespace {{ .Release.Namespace }} port-forward $PROMETHEUS_POD_NAME 9090:$PROMETHEUS_CONTAINER_PORT
{{- end }}
{{- end }}

3. Check the status of your deployment:
   kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "eos-mb-http-sd.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

4. View the logs:
   kubectl logs --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "eos-mb-http-sd.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

5. Check the service discovery endpoint:
   kubectl exec --namespace {{ .Release.Namespace }} -it deployment/{{ include "eos-mb-http-sd.fullname" . }} -- curl http://localhost:8080/sd?job=minio-server
